1//! Utilities for virtual keyboard support
2//!
3//! This module provides you with utilities to handle virtual keyboard instances.
4//! It can be used standalone to implement virtual keyboards or together with
5//! an input method to pass through keys from the keyboard.
6//!
7//! ```
8//! use smithay::{
9//!     delegate_seat, delegate_virtual_keyboard_manager,
10//! #   delegate_compositor
11//! };
12//! use smithay::input::{Seat, SeatState, SeatHandler, pointer::CursorImageStatus};
13//! # use smithay::wayland::compositor::{CompositorHandler, CompositorState, CompositorClientState};
14//! use smithay::wayland::virtual_keyboard::VirtualKeyboardManagerState;
15//! use smithay::reexports::wayland_server::{Display, protocol::wl_surface::WlSurface};
16//! use smithay::input::keyboard::KeyboardHandle;
17//! use smithay::backend::input::{KeyState, Keycode};
18//! use smithay::wayland::virtual_keyboard::VirtualKeyboardHandler;
19//! # use smithay::reexports::wayland_server::Client;
20//! use xkbcommon::xkb::ModMask;
21//!
22//! # struct State { seat_state: SeatState<Self> };
23//!
24//! delegate_seat!(State);
25//! // Delegate virtual keyboard handling for State to VirtualKeyboardManagerState.
26//! delegate_virtual_keyboard_manager!(State);
27//!
28//! # let mut display = Display::<State>::new().unwrap();
29//! # let display_handle = display.handle();
30//!
31//! let seat_state = SeatState::<State>::new();
32//!
33//! // implement the required traits
34//! impl SeatHandler for State {
35//!     type KeyboardFocus = WlSurface;
36//!     type PointerFocus = WlSurface;
37//!     type TouchFocus = WlSurface;
38//!     fn seat_state(&mut self) -> &mut SeatState<Self> {
39//!         &mut self.seat_state
40//!     }
41//!     fn focus_changed(&mut self, seat: &Seat<Self>, focused: Option<&WlSurface>) { unimplemented!() }
42//!     fn cursor_image(&mut self, seat: &Seat<Self>, image: CursorImageStatus) { unimplemented!() }
43//! }
44//!
45//! // Add the seat state to your state, create manager global and add client filter
46//! // to avoid untrusted clients requesting a new keyboard
47//! VirtualKeyboardManagerState::new::<State, _>(&display_handle, |_client| true);
48//!
49//! # impl CompositorHandler for State {
50//! #     fn compositor_state(&mut self) -> &mut CompositorState { unimplemented!() }
51//! #     fn client_compositor_state<'a>(&self, client: &'a Client) -> &'a CompositorClientState { unimplemented!() }
52//! #     fn commit(&mut self, surface: &WlSurface) {}
53//! # }
54//! impl VirtualKeyboardHandler for State {
55//!    fn on_keyboard_event(
56//!     &mut self,
57//!     keycode: Keycode,
58//!     state: KeyState,
59//!     time: u32,
60//!     keyboard: KeyboardHandle<Self>,
61//!  ) { }
62//!    fn on_keyboard_modifiers(
63//!        &mut self,
64//!        depressed_mods: ModMask,
65//!        latched_mods: ModMask,
66//!        locked_mods: ModMask,
67//!        keyboard: KeyboardHandle<Self>,
68//!    ) { }
69//! }
70//! # delegate_compositor!(State);
71//! ```
72//!

92/// State of wp misc virtual keyboard protocol

98/// Data associated with a VirtualKeyboardManager global.
99#[allow(missing_debug_implementations)]
100pub struct VirtualKeyboardManagerGlobalData {
101    filter: Box<dyn for<'c> Fn(&'c Client) -> bool + Send + Sync>,
102}
103
104fn create_global_with_filter<D, F>(display: &DisplayHandle, filter: F) -> GlobalId
105where
106    D: GlobalDispatch<ZwpVirtualKeyboardManagerV1, VirtualKeyboardManagerGlobalData> + 'static,
107    F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
108{
109    let data = VirtualKeyboardManagerGlobalData {
110        filter: Box::new(filter),
111    };
112
113    display.create_global::<D, ZwpVirtualKeyboardManagerV1, _>(MANAGER_VERSION, data)
114}

116/// Handler for the virtual keyboard protocol
117pub trait VirtualKeyboardHandler: SeatHandler {
118    /// Handle KeyboardEvent as usual
119    fn on_keyboard_event(
120        &mut self,
121        keycode: Keycode,
122        state: KeyState,
123        time: u32,
124        keyboard: KeyboardHandle<Self>,
125    );
126
127    /// Handle modifiers event as usual
128    fn on_keyboard_modifiers(
129        &mut self,
130        depressed_mods: ModMask,
131        latched_mods: ModMask,
132        locked_mods: ModMask,
133        keyboard: KeyboardHandle<Self>,
134    );
135}
136
137impl VirtualKeyboardManagerState {
138    /// Initialize a virtual keyboard manager global.
139    pub fn new<D, F>(display: &DisplayHandle, filter: F) -> Self
140    where
141        D: GlobalDispatch<ZwpVirtualKeyboardManagerV1, VirtualKeyboardManagerGlobalData>,
142        D: Dispatch<ZwpVirtualKeyboardManagerV1, ()>,
143        D: Dispatch<ZwpVirtualKeyboardV1, VirtualKeyboardUserData<D>>,
144        D: SeatHandler,
145        D: 'static,
146        F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
147    {
148        let global = create_global_with_filter::<D, F>(display, filter);
149
150        Self { global }
151    }
152
153    /// Get the id of ZwpVirtualKeyboardManagerV1 global
154    pub fn global(&self) -> GlobalId {
155        self.global.clone()
156    }
157}

159impl<D> GlobalDispatch<ZwpVirtualKeyboardManagerV1, VirtualKeyboardManagerGlobalData, D>
160    for VirtualKeyboardManagerState
161where
162    D: GlobalDispatch<ZwpVirtualKeyboardManagerV1, VirtualKeyboardManagerGlobalData>,
163    D: Dispatch<ZwpVirtualKeyboardManagerV1, ()>,
164    D: Dispatch<ZwpVirtualKeyboardV1, VirtualKeyboardUserData<D>>,
165    D: SeatHandler,
166    D: 'static,
167{
168    fn bind(
169        _: &mut D,
170        _: &DisplayHandle,
171        _: &Client,
172        resource: New<ZwpVirtualKeyboardManagerV1>,
173        _: &VirtualKeyboardManagerGlobalData,
174        data_init: &mut DataInit<'_, D>,
175    ) {
176        data_init.init(resource, ());
177    }
178
179    fn can_view(client: Client, global_data: &VirtualKeyboardManagerGlobalData) -> bool {
180        (global_data.filter)(&client)
181    }
182}
183
184impl<D> Dispatch<ZwpVirtualKeyboardManagerV1, (), D> for VirtualKeyboardManagerState
185where
186    D: Dispatch<ZwpVirtualKeyboardManagerV1, ()>,
187    D: Dispatch<ZwpVirtualKeyboardV1, VirtualKeyboardUserData<D>>,
188    D: SeatHandler,
189    D: 'static,
190{
191    fn request(
192        _state: &mut D,
193        _client: &Client,
194        _resource: &ZwpVirtualKeyboardManagerV1,
195        request: zwp_virtual_keyboard_manager_v1::Request,
196        _data: &(),
197        _handle: &DisplayHandle,
198        data_init: &mut DataInit<'_, D>,
199    ) {
200        match request {
201            zwp_virtual_keyboard_manager_v1::Request::CreateVirtualKeyboard { seat, id } => {
202                let seat = Seat::<D>::from_resource(&seat).unwrap();
203                let virtual_keyboard_handle = VirtualKeyboardHandle::default();
204                data_init.init(
205                    id,
206                    VirtualKeyboardUserData {
207                        handle: virtual_keyboard_handle,
208                        seat: seat.clone(),
209                    },
210                );
211            }
212            _ => unreachable!(),
213        }
214    }
215}

219macro_rules! delegate_virtual_keyboard_manager {
220    ($(@<$( $lt:tt $( : $clt:tt $(+ $dlt:tt )* )? ),+>)? $ty: ty) => {
221        const _: () = {
222            use $crate::{
223                reexports::{
224                    wayland_protocols_misc::zwp_virtual_keyboard_v1::server::{
225                        zwp_virtual_keyboard_manager_v1::ZwpVirtualKeyboardManagerV1,
226                        zwp_virtual_keyboard_v1::ZwpVirtualKeyboardV1,
227                    },
228                    wayland_server::{delegate_dispatch, delegate_global_dispatch},
229                },
230                wayland::virtual_keyboard::{
231                    VirtualKeyboardManagerGlobalData, VirtualKeyboardManagerState, VirtualKeyboardUserData,
232                },
233            };
234
235            delegate_global_dispatch!(
236                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
237                $ty: [ZwpVirtualKeyboardManagerV1: VirtualKeyboardManagerGlobalData] => VirtualKeyboardManagerState
238            );
239
240            delegate_dispatch!(
241                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
242                $ty: [ZwpVirtualKeyboardManagerV1: ()] => VirtualKeyboardManagerState
243            );
244
245            delegate_dispatch!(
246                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
247                $ty: [ZwpVirtualKeyboardV1: VirtualKeyboardUserData<Self>] => VirtualKeyboardManagerState
248            );
249        };
250    };
